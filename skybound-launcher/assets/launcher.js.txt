// Load sounds
const launchSound = new Audio('assets/launch.mp3');
const hoverSound = new Audio('assets/hover.mp3');

// Play button
document.querySelector('.play-button').addEventListener('click', () => {
  launchSound.play();
  alert('Launching Skybound Syndicate...');
});

// Hover sounds
document.querySelectorAll('nav button, .play-button').forEach(button => {
  button.addEventListener('mouseenter', () => {
    hoverSound.play();
  });
});

// Versions menu toggle
const versionsButton = document.getElementById('versions-button');
const versionsMenu = document.getElementById('versions-menu');

versionsButton.addEventListener('click', () => {
  versionsMenu.classList.toggle('hidden');
});

// Version selection
document.querySelectorAll('#versions-menu li').forEach(item => {
  item.addEventListener('click', () => {
    const version = item.getAttribute('data-version');
    alert(`Launching version: ${version}`);
  });
});

// Load remembered user
const savedUser = localStorage.getItem('skyboundUser');
if (savedUser) {
  document.getElementById('user-name').textContent = savedUser;
}
